       ```typescript
       export function getRecencyColor(lastActivityDate: string | Date): 'success' | 'warning' | 'error' {
         const now = new Date()
         const last = new Date(lastActivityDate)
         const diffMs = now.getTime() - last.getTime()
       ```

       ```typescript
       import { useState, useEffect } from 'react'

       export function useDebounce<T>(value: T, delay: number): T {
         const [debouncedValue, setDebouncedValue] = useState(value)

         useEffect(() => {
           const handler = setTimeout(() => setDebouncedValue(value), delay)
           return () => clearTimeout(handler)
         }, [value, delay])

         return debouncedValue
       }
       ```

    7. Add `groupByDay` utility to `src/lib/utils.ts` (used by BFIX-08, AFIX-01):
       ```typescript
       import { isToday, isYesterday, format } from 'date-fns'

       export interface DayGroup<T> {
         label: string
         date: Date
         items: T[]
       }

       export function groupByDay<T extends { created_at: string }>(items: T[]): DayGroup<T>[] {
         const groups = new Map<string, T[]>()
         for (const item of items) {
           const date = new Date(item.created_at)
           const key = format(date, 'yyyy-MM-dd')
           const existing = groups.get(key) ?? []
           existing.push(item)
           groups.set(key, existing)
         }
         return Array.from(groups.entries()).map(([key, groupItems]) => {
           const date = new Date(key)
           let label: string
           if (isToday(date)) label = 'Today'
           else if (isYesterday(date)) label = 'Yesterday'
           else label = format(date, 'MMMM d, yyyy')
           return { label, date, items: groupItems }
         })
       }
       ```
