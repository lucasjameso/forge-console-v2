   import { subDays, startOfDay, isAfter, isToday, isYesterday } from 'date-fns'

   const filteredByDate = entries.filter(entry => {
     const entryDate = new Date(entry.created_at)
     switch (dateRange) {
       case 'today': return isToday(entryDate)
